Simple to Integrate

Start fetching products in under 5 minutes.

// Fetch products from GlobexSky API const API_KEY = 'sk_live_your_api_key_here'; async function fetchProducts(query, page = 1) { const response = await fetch( `https://api.globexsky.com/v1/products?q=${query}&page=${page}`, { method: 'GET', headers: { 'Authorization': `Bearer ${API_KEY}`, 'Content-Type': 'application/json' } } ); if (!response.ok) throw new Error(`API error: ${response.status}`); const { data, meta } = await response.json(); console.log(`Found ${meta.total} products`); return data; } // Example usage fetchProducts('wireless headphones') .then(products => console.log(products)) .catch(err => console.error(err));
